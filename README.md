dotfiles
========

install all:

    git clone git@github.com:gerardorochin/dotfiles.git ~/.dotfiles
    cd ~/.dotfiles
    ./bootstrap.sh

run standalone script:

    ./scripts/SCRIPTNAME.sh

force:

    ./bootstrap.sh -f
